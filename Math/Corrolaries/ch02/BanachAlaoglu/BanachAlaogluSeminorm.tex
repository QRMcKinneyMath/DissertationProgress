\begin{cor}[Banach Alaoglu Seminorm]
\label{cor:BanachAlaogluSeminorm}
\rm
Let $X$ be a \SeminormedSpace.
Let $B=\{x^* \in X^*: \norm{x^*} \leq 1\}$. 
Then $B$ is $\weakstar$ \SetCompact. 
\begin{proof} 
Define $\tilde{X} = X/\Ker_X$. 
Define $\tilde{B} = \{ x^* \in \tilde{X}^* : \norm{x^*} \leq 1 \}$.
By
\ref{thm:BanachAlaoglu}, 
$\tilde{B}$ is \weakstar \SetCompact. 
By 
\ref{def:higherorderdualspaceisomorphism}
$\tilde{B}$ is \weakstar isomorphic to $B$,
an so $B$ is \weakstar \SetCompact.
\end{proof} 
\end{cor}
