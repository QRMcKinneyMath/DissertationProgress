\begin{cor}[Banach-Alaoglu-Morales]
\label{thm:BanachAlaogluMorales}
\rm
Let $X$ be a \SeminormedSpace.
Let $C \subset X^*$.
The following are true. 
\begin{enumerate}
\item 
\label{cor:BanachAlaoglu:CompleteCompactImpliesClosedAndBounded}
If $X$ is \Complete and $C$ is 
\weakstar \SetCompact, then $C$ is \weakstar \SetClosed and 
\Norm bounded.
\item 
\label{cor:BanachAlaoglu:ClosedAndBoundedImpliesCompact}
If $C$ is \weakstar \SetClosed and 
\Norm bounded, then $C$ is \weakstar \SetCompact.
\end{enumerate}
\begin{proof}[Proof of \ref{cor:BanachAlaoglu:CompleteCompactImpliesClosedAndBounded}]
Since $C$ is \weakstar \SetCompact
and the \weakstar \Topology is \Hausdorff, 
$C$ is \weakstar \SetClosed.
Let $x \in X$.
Then $c(x):(X,\mathcal{T}_{w^*}) \to \F$ is \ContinuousFunction, so
$c(x)(C)=$ is \SetCompact and therefore bounded in $\F$.
Since $x \in X$ was arbitrary, by the Banach-Steinhauss theorem, 
$C$ is \Norm bounded in $X$.
\end{proof}
\begin{proof}[Proof of \ref{cor:BanachAlaoglu:ClosedAndBoundedImpliesCompact}]
Since $C$ is \Norm bounded, for some $0 \neq \alpha \in \F$, $\alpha C \subset B_{X^*}(0;1)$.
Since scaling by $\alpha$ is a \Homeomorphism, $\alpha C$ is \weakstar \SetClosed.
By
\ref{cor:BanachAlaogluSeminorm}, 
$\alpha C$ is a \weakstar \SetClosed subset of a \weakstar \SetCompact 
set and is therefore itself \weakstar \SetCompact. 
Hence $C$ is \weakstar \SetCompact.
\end{proof} 
\end{cor}
