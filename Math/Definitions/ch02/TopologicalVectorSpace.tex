\label{def:topologicalvectorspace}
\newcommand{\TVS}[0]{
    \bf \hyperref[def:topologicalvectorspace]{Topological Vector Space} \rm
}

\begin{df}[\TVS]
Let $(V,+,\cdot, 0)$ be a 
\VectorSpace over a \Field $\F \in \{\R, \C\}$. 
Let $\T$ be a 
\Topology on $V$
which is 
\VectorSpaceCompatible
with $(V, + , \cdot, 0)$. 
Then we call 
$(V,\T)$ a 
\TVS.
\end{df}
