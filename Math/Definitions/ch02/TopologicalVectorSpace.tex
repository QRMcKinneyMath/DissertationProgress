\newcommand{\TVS}[0]{
    \bf \hyperref[def:topologicalvectorspace]{Topological Vector Space} \rm
}

\begin{df}[\TVS]
Let $(V,+,\cdot, 0)$ be a 
\VectorSpace over a \Field $\F \in \{\R, \C\}$. 
Let $\T$ be a 
\TopologyRef on V such that 
$(V, +, 0)$ is a 
\TopologicalGroup
and 
$\cdot: \F \times V \to V$ 
is
\Continuous. 
Then we call $(V,\T)$ a \TVS. 
\end{df}
