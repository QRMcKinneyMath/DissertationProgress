\newcommand{\Chains}[0]{\textbf{\hyperref[def:Chain]{Chains}}\xspace}
\begin{df}[\Chain]
    Let $(X,\leq)$ be a \Poset. 
    Let $A \subset X$ such that
    $\pa{A, \leq \cap \pa{A \times A}}$ is a 
    \Toset. 
    Then we call $A$ a \Chain in $X$. 
\end{df}
