\newcommand{\StrictlyConvexSpace}[0]{\textbf{\hyperref[def:StrictlyConvexSpace]{Strictly Convex}}\xspace}
\begin{df}[\StrictlyConvexSpace \SeminormedSpace]
\label{def:StrictlyConvexSpace}
\rm
We say that a \SeminormedSpace $(X, \norm{\cdot})$ is 
\StrictlyConvexSpace if for each $x,y \in \partial B_X(0;1)$ such that 
$x \not \in y + \overline{0}$, for each $\lambda  \in (0,1)$, 
\begin{equation*}
\norm{\lambda x + (1-\lambda) y } < 1
\end{equation*}
\end{df}
