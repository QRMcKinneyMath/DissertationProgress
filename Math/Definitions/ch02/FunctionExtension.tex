\newcommand{\Extensions}[0]{\textbf{\hyperref[def:FunctionExtension]{Extensions}}\xspace}
\begin{df}[\Extension]
\label{def:FunctionExtension}

\rm
    Let $X$ and $Y$ be nonempty sets.
	Let $g:X \to Y$. 
	Let $f$ be a \Restriction of $g$.  
    Then we call $g$ an 
    \Extension of $f$. 
\end{df}
