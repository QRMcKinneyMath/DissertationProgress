\newcommand{\LocalConvexity}[0]{
    \bf \hyperref[def:LocallyConvex]{Local Convexity} \rm
}

\begin{df}[\LocallyConvex]
    We say that a 
    \TVS
    $(X,\T)$ is 
    \LocallyConvex 
    if $(X,\T)$ has a 
    \LocalBasis consisting only of 
    \ConvexSet sets.
    A \LocallyConvex 
    space is said to posess
    \LocalConvexity.
\end{df}
