\begin{df}[\GreatestLowerBound]
    Let $X \neq \emptyset$ be a set. 
    Let $R$ be a \Relation on X. 
    Let $Y \subset X$.
	Let $a \in X$. 
	We say that $a$ is a
	\GreatestLowerBound of $Y$ if 
	$a \in \Maxima(\LB(Y))$.
	We denote the set of \GreatestLowerBounds
	for $Y$ with $\GLB(Y)$.
	If $b \in \GLB(Y)$, then we 
	also call $b$ a 
	\Infimum of $Y$. 
	The Plural of \Infimum is \Infima. 
	If $\GLB(Y)=\{c\}$, then 
	we write $c=\Inf(Y)$.  
\end{df}
