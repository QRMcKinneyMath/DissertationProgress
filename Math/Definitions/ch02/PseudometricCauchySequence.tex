\begin{df}[Pseudometric Cauchy Sequence]
\label{def:pseudometriccauchysequence}
\rm
    Let $(X,d)$ be a \PseudometricSpace.
    We say that a sequence 
	$\{x_i\}_{i \in \N}$ is a 
	\PseudometricCauchySequence
    if, for each 
	$\epsilon > 0$, 
	there exists
	$N \in \N$
	such that for 
    each pair 
	$m,n \in \N$ 
	such that 
	$m>N$ 
	and 
	$n>N$, we have 
    \begin{equation*}
        d(x_m,x_n) < \epsilon
    \end{equation*}
\end{df}
