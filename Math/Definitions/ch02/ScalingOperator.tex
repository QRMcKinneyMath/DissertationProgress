\label{def:ScalingOperator}
\newcommand{\ScalingOperator}[0] {
    \bf \hyperref[def:ScalingOperator]{Scaling Operator} \rm
}

\begin{df}[\ScalingOperator]
    Let $V$ be a 
    \VectorSpace 
    over a 
    \Field $\F$. 
    Let $\alpha \in \F$. 
    We define $M_\alpha:V \to V$ by 
    setting, for each 
    $x \in V$, 
    \begin{equation}
    M_\alpha(x)=\alpha x
    \end{equation}
    We call $M_\alpha$ the
    \ScalingOperator
\end{df}
