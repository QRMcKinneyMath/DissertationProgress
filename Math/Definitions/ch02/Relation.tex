\newcommand{\Relation}[0]{
    \bf \hyperref[def:Relation]{Relation} \rm
}
\begin{df}[\Relation]
    Let $X \neq \emptyset$ be a set. 
    We say that $R$ is a \Relation
    on X if $R \subset X \times X$. 
    If $(a,b) \in R$, then we may write
    $a R b$. 
\end{df}
