\begin{df}[Relation]
\label{def:Relation}

\rm
    Let $X \neq \emptyset$ be a set
	and let $Y\neq \emptyset$ be a set. 
    We say that $R$ is a \Relation
    from $X$ to $Y$ if $R \subset X \times Y$. 
    If $(a,b) \in R$, then we may write
    $a R b$. 
\end{df}
