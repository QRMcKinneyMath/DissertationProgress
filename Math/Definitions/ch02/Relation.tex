\newcommand{\Relation}[0]{\textbf{\hyperref[def:Relation]{Relation}}\xspace}
\begin{df}[\Relation]
\label{def:Relation}

\rm
    Let $X$ and $Y$ be nonempty sets.
    We say that $R$ is a \Relation
    from $X$ to $Y$ if $R \subset X \times Y$. 
    If $(a,b) \in R$, then we write
    $a R b$. 
    If $X=Y$ then we call $R$ a 
    \Relation on $X$.
\end{df}
