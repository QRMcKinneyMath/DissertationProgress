\begin{df}[\LeftInverseElement, \RightInverseElement]
    Let $X \neq \emptyset$ be a set 
    and let $\oplus$ be a 
    \BinaryOperation on $X$. 
    Let $e$ be an \IdentityElement
    for $\oplus$. 
    Let $l,r \in X$ such that 
    \begin{equation}
        l \oplus r=e
    \end{equation}
    In this scenario, we say that 
    $l$ is a 
    \LeftInverseElement 
    of $r$  under the \Operation
    $\oplus$
    and we say that 
    $r$
    is a 
    \RightInverseElement
    of $l$ 
    under the \Operation
    $\oplus$. 
    Furthermore, we say that 
    $r$ is 
    \LeftInvertibleElement
    with respect to the operation 
    $\oplus$
    and that 
    $l$ is 
    \RightInvertibleElement 
    with arespect to the operation
    $\oplus$. 
    If the \Operation
    is clear, we may of course omit it from what we say. 
\end{df}

\begin{df}[\InverseElement]
    Let $X \neq \emptyset$ 
    and $\oplus$ be a 
    \BinaryOperation on $X$. 
    Let $e$ be an 
    \IdentityElement
    of $\oplus$. 
    Let $x,y \in X$ such that
    $x$ is a \LeftInverseElement
    of $y$
    and $x$
    is a 
    \RightInverseElement
    of $y$. 
    In such a scenario, we say that 
    $x$ is an 
    \InverseElement
    of $y$
    with respect to the 
    \Operation
    $\oplus$
    and we call 
    $y$ an 
    \InvertibleElement
    element
    of $X$ 
    with respect to the
    \Operation $\oplus$. 
\end{df}
