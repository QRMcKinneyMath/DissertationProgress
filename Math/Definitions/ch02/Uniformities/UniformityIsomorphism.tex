\begin{df}[\UniformityIsomorphism]
    \label{def:UniformityIsomorphism}
    Let $X_1,X_2 \neq \emptyset$.
    Let $\scW_i$ be a \Uniformity on 
    $X_i$ for $i \in \{1,2\}$. 
    We say that 
    $f:X_1 \to X_2$ is a \UniformityIsomorphism
    of $(X_1, \scW_1)$ onto $(X_2, \scW_2)$  
    if $f\times f (\scW_1) = \scW_2$, 
    where $f \times f$ denotes the 
    \FunctionProduct. 
    
\end{df}
