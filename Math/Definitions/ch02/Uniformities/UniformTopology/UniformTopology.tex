\begin{df}[\UniformTopology]
\label{def:UniformTopology}
Let $X \neq \emptyset$ and let 
$\scW$ be a \Uniformity on $X$.
Let $\scT_{\scW}$ be the \Topology defined in
\ref{prop:UniformTopology}.
Then we call $\scT_{\scW}$ the 
\UniformTopology on $X$ 
\UniformInduced by $\scW$. 
We may call $(X, \scW, \scT_{\scW})$ a \UniformSpace.
\end{df}
