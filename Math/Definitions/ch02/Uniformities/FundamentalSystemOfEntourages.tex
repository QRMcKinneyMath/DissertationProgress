\newcommand{\FundamentalSystemsOfEntourages}[0]{\textbf{\hyperref[def:FundamentalSystemOfEntourages]{Fundamental Systems Of Entourages}}\xspace}
\begin{df}[\FundamentalSystemOfEntourages]
    \label{def:FundamentalSystemOfEntourages}
    Let $X \neq \emptyset$ and 
    \scW be a \Uniformity on X. 
    Let $\scB \subset \scW$ such that 
	\scNested{\scB, \scW} holds.
    Then we call 
    \scB 
    a 
    \FundamentalSystemOfEntourages for \scW.
\end{df}
