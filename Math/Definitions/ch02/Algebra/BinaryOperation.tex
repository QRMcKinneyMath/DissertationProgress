\begin{df}[\Operation, \UnaryOperation, \BinaryOperation]
    Let $X \neq \emptyset$
    be a set. 
    Let $A \neq \emptyset$
    be a set with
    $cardinality(A)=n \in \N$. 
    We call a mapping 
    \begin{equation*}
        T:\prod\limits_{\alpha \in A} X \to X
    \end{equation*}
    an 
    n-ary \Operation
    on $X$. 
    If $n=1$ 
    then we call $T$ a
    \UnaryOperation
    on 
    $X$. 
    If $n=2$, 
    then we call $T$ a 
    \BinaryOperation
    on $X$. 
    If $T$ is a 
    \BinaryOperation
    on $X$, 
    we sometimes use the notation
    \begin{equation*}
        xTy=T(x,y)
    \end{equation*}
\end{df}
