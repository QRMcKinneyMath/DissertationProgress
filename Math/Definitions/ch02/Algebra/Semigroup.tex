\begin{df}[\Semigroup]
    Let $(X,\oplus)$ be a 
    \Magma.
    Let $\oplus$ 
    be 
    \AssociativeFunction.
    Then we say that 
    $(X,\oplus)$ 
    is a 
    \Semigroup. 
\end{df}
