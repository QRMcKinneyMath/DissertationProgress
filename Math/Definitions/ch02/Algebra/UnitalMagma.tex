\begin{df}[\UnitalMagma]
\label{def:UnitalMagma}
\rm
    Let $(X,\oplus)$ be a
    \Magma with \IdentityElement $e$. 
    Then we call 
    $(X,\oplus,e)$ a
    \UnitalMagma.
\end{df}
