\begin{df}
\label{def:Subgroup}
\rm
Let $G$ be a \Group.
Let $H \subset G$ be \SymmetricSubset.
Furthermore, suppose 
$HH \subset H$.
Then we call $H$ a \Subgroup of $G$. 
If $K$ is a \Subgroup
of $G$ and for each $x \in G$, 
$xKx^{-1} = K$, 
then we call $K$ a 
\NormalSubgroup of $G$.
\end{df}
