\newcommand{\SymmetricMap}[0]{\textbf{\hyperref[def:Symmetricmap]{Symmetric Map}}\xspace}
\newcommand{\CommutativeFunction}[0]{\textbf{\hyperref[def:Symmetricmap]{Commutative}}\xspace}
\newcommand{\FunctionCommutativity}[0]{\textbf{\hyperref[def:Symmetricmap]{Commutativity}}\xspace}
\begin{df}[\CommutativeFunction]
\label{def:Symmetricmap}
\rm
    Let X and Y be nonempty sets. 
    We say that a map 
    $f:X \times X \to Y$ is a \SymmetricMap 
    if for each 
    $x_0,x_1 \in X$, 
    $f(x_0,x_1)=f(x_1,x_0)$.
    In this situation, 
    we may also refer to $f$ as
    \CommutativeFunction, 
    or say that $f$ posesses 
    \FunctionCommutativity.
\end{df}
