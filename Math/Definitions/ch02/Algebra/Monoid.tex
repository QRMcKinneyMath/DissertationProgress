\begin{df}[\Monoid]
    Let $(X,\oplus,e)$ be a 
    \UnitalMagma
    and let
    $(X,\oplus)$ 
    be a \Semigroup. 
    Then we call
    $(X,\oplus,e)$ 
    a 
    \Monoid.
\end{df}
