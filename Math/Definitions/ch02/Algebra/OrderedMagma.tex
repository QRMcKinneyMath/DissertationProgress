\begin{df}[\PartiallyOrderedMagma, \TotallyOrderedMagma, \DirectedMagma]
    Let $(X,\oplus)$ 
    be a 
    \Magma.
    Let $T$ be a 
    \TotalOrdering 
    on $X$ 
    which is 
    \AlgebraicallyConsistent
    with $(X,\oplus)$. 
    Let $P$ be a 
    \PartialOrdering 
    on $X$
    which is 
    \AlgebraicallyConsistent
    with $(X,\oplus)$. 
    Let $D$ be a 
    \Directing 
    on $X$
    which is 
    \AlgebraicallyConsistent
    with $(X,\oplus)$. 
    We call 
    $(X,\oplus, T)$ a 
    \TotallyOrderedMagma.
    $(X,\oplus, P)$ a 
    \PartiallyOrderedMagma.
    $(X,\oplus, D)$ a 
    \DirectedMagma.
\end{df}
