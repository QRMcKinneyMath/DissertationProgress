\begin{df}[\PartiallyOrderedMagma, \TotallyOrderedMagma, \DirectedMagma]
\label{def:OrderedMagma}
\rm
    Let $(X,\oplus)$ 
    be a 
    \Magma.
    Let $T$ be a 
    \TotalOrdering 
    on $X$ 
    which is 
    \AlgebraicallyConsistent
    with $(X,\oplus)$. 
    Let $P$ be a 
    \PartialOrdering 
    on $X$
    which is 
    \AlgebraicallyConsistent
    with $(X,\oplus)$. 
    Let $D$ be a 
    \Directing 
    on $X$
    which is 
    \AlgebraicallyConsistent
    with $(X,\oplus)$. 
    We call 
    $(X,\oplus, T)$ a 
    \TotallyOrderedMagma.
    we call $(X,\oplus, P)$ a 
    \PartiallyOrderedMagma, and we call
    $(X,\oplus, D)$ a 
    \DirectedMagma.
\end{df}
