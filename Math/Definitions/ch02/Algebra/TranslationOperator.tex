\begin{df}[\TranslationOperator]
\label{def:TranslationOperator}
\rm
    Let $G$ be a 
    \Magma. 
    Let $g \in G$. 
    We define 
    $\scRightTranslationOperator_g:G \to G$
    and
    $\scLeftTranslationOperator_g:G \to G$
    by setting, for each 
    $x \in G$, 
    \begin{equation*}
        \scRightTranslationOperator_g (x) = xg
    \end{equation*}
    \begin{equation*}
        \scLeftTranslationOperator_g (x) = gx 
    \end{equation*}
    We call $\scRightTranslationOperator_g$
    the \RightTranslation
    of $G$
    by $g$, 
    and we call 
    $\scLeftTranslationOperator_g$
    the \LeftTranslation
    of 
    $G$ 
    by $g$. 
    If the operation of $G$ is 
    \CommutativeFunction,
    then we define 
    $\scTranslationOperator_g=\scRightTranslationOperator_g=\scLeftTranslationOperator_g$
    which we call 
    \Translation of $G$ by $g$.
\end{df}
