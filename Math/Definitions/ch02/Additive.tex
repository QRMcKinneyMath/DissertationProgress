\newcommand{\MagmaHomomorphism}[0]{
    \bf \hyperref[def:additive]{Magma Homomorphism} \rm
}
\begin{df}[\Additive]
    Let $G$ and $H$ 
    be a Magmas.
    Represent the operation of $G$ and 
    $H$ both with $+$. 
    We call a mapping $p:G \to H$ 
    a \MagmaHomomorphism 
    if, for every 
    $x,y \in G$, we have 
    \begin{equation}
        p(x+y) = p(x)+p(y)
    \end{equation}
    We say that a 
    \MagmaHomomorphism
    is an 
    \Additive 
    operator
    and that 
    \Additive
    operators possess
    the property \Additivity
\end{df}
