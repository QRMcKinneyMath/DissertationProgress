\newcommand{\UniformlySmoothSpace}[0]{\textbf{\hyperref[def:UniformlySmoothSpace]{Uniformly Smooth}}\xspace}
\begin{df}[\UniformlySmoothSpace]
\label{def:UniformlySmoothSpace}
\rm
Let $X$ be a \SeminormedSpace. 
Let $\rho$ denote the \ModulusOfSmoothness of $X$. 
We say that $X$ is \UniformlySmoothSpace if 
\begin{equation*}
\lim\limits_{\epsilon \to 0} \frac{\rho(\epsilon)}{\epsilon} = 0
\end{equation*}
\end{df}
