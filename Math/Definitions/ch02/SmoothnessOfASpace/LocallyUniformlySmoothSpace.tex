\newcommand{\LocallyUniformlySmoothSpace}[0]{\textbf{\hyperref[def:LocallyUniformlySmoothSpace]{Locally Uniformly Smooth}}\xspace}
\begin{df}[\LocallyUniformlySmoothSpace]
\label{def:LocallyUniformlySmoothSpace}
\rm
Let $X$ be a \SeminormedSpace. 
Let $x_0 \in X$. 
Let $\overline{\rho}(\cdot, x_0)$ denote the 
\ModulusOfSmoothness of $X$ at $x_0$. 
Suppose
\begin{equation*}
\lim\limits_{\epsilon \to 0} \frac{\overline{\rho}(\epsilon, x_0)}{\epsilon} = 0
\end{equation*}
Then we say that $X$ is \LocallyUniformlySmoothSpace at $x_0$. 
If $X$ is \LocallyUniformlySmoothSpace at each of its points, 
then we say that $X$ is \LocallyUniformlySmoothSpace.
\end{df}
