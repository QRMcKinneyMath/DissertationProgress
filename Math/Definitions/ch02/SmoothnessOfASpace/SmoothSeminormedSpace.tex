\newcommand{\SmoothSpace}[0]{\textbf{\hyperref[def:SmoothSpace]{Smooth}}\xspace}
\begin{df}[\SmoothSpace]
\label{def:SmoothSpace}
\rm
Let $X$ be a \SeminormedSpace.
Let $J$ be the \NormalizedDualityMap of $X$. 
Let $x_0 \in X$.
We say that $X$ is \SmoothSpace
at $x_0$ if $J(x_0)$ is a singleton.
We say that $X$ is \SmoothSpace
if $X$ is \SmoothSpace at each $x \in X$.
\end{df}
