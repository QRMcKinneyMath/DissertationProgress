\newcommand{\NormalizedDualityMap}[0]{\textbf{\hyperref[def:NormalizedDualityMap]{Normalized Duality Map}}\xspace}
\begin{df}[\NormalizedDualityMap]
\label{def:NormalizedDualityMap}
\rm
Let $X$ be a \SeminormedSpace.
Define $J:X \to 2^{X^*}$ by 
\begin{equation*}
    Jx=\left\{x^* \in X^* : \ip{x,x^*}=\norm{x}^2=\norm{x^*}^2\right\}
\end{equation*} 
We call $J$ the \NormalizedDualityMap of $X$.
\end{df}
