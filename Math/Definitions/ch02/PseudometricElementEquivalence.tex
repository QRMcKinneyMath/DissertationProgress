\begin{df}[Relation Of Zero Distance]
    Let $(X,d)$ be a \PseudometricSpace. 
    Define the relation  $\cong_d$ on $X \times X$ by setting, for $x,y \in X$, 
    \begin{equation}
        x \cong_d y \iff d(x,y) = 0
    \end{equation}
    We call $\cong_d$ the \RelationOfZeroDistance on $(X,d)$. 
\end{df}
