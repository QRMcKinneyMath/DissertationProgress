\newcommand{\PreorderedSet}[0]{\textbf{\hyperref[def:Preorder]{Preordered Set}}\xspace}

\begin{df}[\Preorder]
\label{def:Preorder}
\rm
    Let $X$ be a nonempty set.
    Let $R$ be a 
    \TransitiveRelation, 
    \ReflexiveRelation
    \Relation on $X$. 
    Then we call $R$
    a \Preorder on $X$, 
    we call
    $R$ a \Preordering 
    of $X$, and we call 
    $(X,R)$ a \PreorderedSet.
    \end{df}
