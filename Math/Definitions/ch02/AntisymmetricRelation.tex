\begin{df}[\AntiSymmetricRelation]
    Let $X \neq \emptyset$ be a set. 
    Let $R$ be a \Relation on X. 
    We say that $R$ is \AntiSymmetricRelation, 
    or equivalently we say that
    $R$ posseses 
    \RelationAntiSymmetry
    if whenever $aRb$ and $bRa$, we 
    must have $a = b$.
\end{df}
