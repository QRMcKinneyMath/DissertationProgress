\begin{df}[Dual Space]
    Let $(X,\norm{\cdot})($ be a 
    \SeminormedSpace.
    We call $BL(X, \F)$ the 
    \TopDualSpace of 
    $(X,\norm{\cdot})$,
    and we denote 
    $BL(X,\F)$ with the symbol 
    $X^*$. 
    If $x^* \in X^*$, then
    we use the notational convention
    of writing, for $x \in X$. 
    \begin{equation}
    \ip{x, x^*}:= x^*(x)
    \end{equation}
\end{df}
\begin{rmk}[\TopDualSpace is a \NormedSpace]
    Let $X$ be a 
    \SeminormedSpace.
    Then, using 
    \ref{prop:BLO}, 
    since $\F$ is a 
    \NormedSpace, 
    so is $X^*$. 
\end{rmk}
