\begin{df}[\ClosedFunction]
    Let $(X,\T_X)$ and
    $(Y, \T_Y)$ 
    be 
    \TopologicalSpaces. 
    We say that 
    $f:X \to Y$ 
    is 
    \ClosedFunction
    if 
    $f(K)$ is 
    \SetClosed 
    in $(Y,\T_Y)$ 
    for every 
    \SetClosed
    $K \in (X,\T_X)$. 
\end{df}
