\begin{df}[\DiscreteTopology, \IndiscreteTopology]
    Let $X$ be a set. 
    We call $\{X, \emptyset\}$ 
    the \IndiscreteTopology
    on $X$
    and we call 
    \scPowerSet{X}
    the 
    \DiscreteTopology 
    on $X$. 
\end{df}
