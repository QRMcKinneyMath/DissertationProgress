\newcommand{\TopologyBasis}[0]{\textbf{\hyperref[def:TopologyBasis]{Basis}}\xspace}
\newcommand{\TopologyBases}[0]{\textbf{\hyperref[def:TopologyBasis]{Bases}}\xspace}
\begin{df}[\TopologyBasis]
    Let $(X,\T)$ 
    be a 
    \TopologicalSpace
    and let $B \subset \T$ 
    such that 
    each element of 
    $\T$ 
    can be written as a union 
    of elements of 
    $B$. 
    Then we call 
    $B$
    a 
    \TopologyBasis
    for $\T$. 
\end{df}
