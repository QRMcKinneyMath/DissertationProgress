\begin{df}[\TopologyDense]
    Let $(X,\T)$ be a 
    \TopologicalSpace
    and let $A \subset X$. 
    We say that $A$ is
    \TopologyDense
    in $X$ if 
    $\ClosureMark(A) = X$. 
\end{df}
