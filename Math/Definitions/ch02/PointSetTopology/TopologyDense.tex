\newcommand{\TopologyDensity}[0]{\textbf{\hyperref[def:TopologyDense]{Density}}\xspace}
\begin{df}[\TopologyDense]
\label{def:TopologyDense}
\rm
    Let $(X,\T)$ be a 
    \TopologicalSpace
    and let $A \subset X$. 
    We say that $A$ is
    \TopologyDense
    in $X$ if 
    $\ClosureMark{A} = X$. 
\end{df}
