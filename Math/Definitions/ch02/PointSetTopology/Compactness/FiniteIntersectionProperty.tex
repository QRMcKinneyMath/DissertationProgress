\newcommand{\FiniteIntersectionProperty}[0]{\textbf{\hyperref[def:FiniteIntersectionProperty]{Finite Intersection Property}}\xspace}
\begin{df}[\FiniteIntersectionProperty]
\label{def:FiniteIntersectionProperty}
\rm
Let $X=\{X_\alpha\}_{\alpha \in A}$ be a collection of sets.
Suppose that for every \Finite $K \subset A$, we have 
\begin{equation*}
\emptyset \neq \bigcap\limits_{k \in K} X_k
\end{equation*}
Then we say that $X$ possesses the \FiniteIntersectionProperty.
\end{df}
