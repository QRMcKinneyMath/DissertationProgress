\begin{df}[\CompactFunction]
    Let $(X,\T_X)$ and
    $(Y, \T_Y)$ 
    be 
    \TopologicalSpaces. 
    We say that 
    $f:X \to Y$ 
    is 
    \CompactFunction
    if 
    $f(K)$ is 
    \SetCompact 
    in $(Y,\T_Y)$ 
    for every 
    \SetCompact
    $K \in (X,\T_X)$. 
\end{df}
