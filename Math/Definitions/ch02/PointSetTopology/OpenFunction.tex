\newcommand{\FunctionOpenness}[0]{\textbf{\hyperref[def:OpenFunction]{Openness}}\xspace}
\begin{df}[\OpenFunction]
\label{def:OpenFunction}

\rm
    Let $(X,\T_X)$ and
    $(Y, \T_Y)$ 
    be 
    \TopologicalSpaces. 
    We say that 
    $f:X \to Y$ 
    is 
    \OpenFunction
    if 
    $f(U)$ is 
    \SetOpen 
    in $(Y,\T_Y)$ 
    for every 
    \SetOpen
    $U \in (X,\T_X)$. 
\end{df}
