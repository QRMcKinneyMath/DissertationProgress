\begin{df}[\TopologySubBasis]
    Let $X \neq \emptyset$ 
    and let $B \subset \scPowerSet{X}$. 
    We denote the 
    \TopologyCoarsest
    \TopologyRef on $X$ 
    containing $B$
    with
    \scGeneratedTopology{X}{B}.
    We say that $B$
    is a 
    \TopologySubBasis
    for 
    \scGeneratedTopology{X}{B}
    and we call 
    \scGeneratedTopology{X}{B}
    the 
    \TopologyRef
    on $X$ 
    \TopologyGeneratedBy
    $B$. 
\end{df}
