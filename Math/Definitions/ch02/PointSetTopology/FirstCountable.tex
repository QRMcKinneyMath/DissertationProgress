\begin{df}[\FirstCountable]
\label{def:FirstCountable}
\rm
    Let $(x,\T)$ be a 
    \TopologicalSpace. 
    We say that 
    $X$
    is 
    \FirstCountable
    if for each $x \in X$, 
    there is a 
    \Countable
    \NeighborhoodBasis
    for 
    $\T$
    at
    $X$. 
\end{df}
