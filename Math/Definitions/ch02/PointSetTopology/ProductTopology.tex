\newcommand{\ProductTopologies}[0]{\textbf{\hyperref[def:ProductTopology]{Product Topologies}}\xspace}

\begin{df}[\ProductTopology]
    Let $A \neq \emptyset$. 
    For each $\alpha \in A$, 
    let $(X_\alpha, \T_\alpha)$ 
    be a 
    \TopologicalSpace.
    We call the \WeakTopology
    on 
    \scCartesianProduct{\alpha}{A}{X}
    induced by 
    $\{\pi_\alpha:\scCartesianProduct{\alpha}{A}{X} \to (X_\alpha, \T_\alpha)\}_{\alpha \in A}$
    the \ProductTopology. 
\end{df}
