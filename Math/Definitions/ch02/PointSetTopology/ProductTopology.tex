\begin{df}[\ProductTopology]
\label{def:ProductTopology}

\rm
    Let $A$ be a nonempty set. 
    For each $\alpha \in A$, 
    let $(X_\alpha, \T_\alpha)$ 
    be a 
    \TopologicalSpace.
	For each $\alpha \in A$, let 
	$\pi_\alpha:\scCartesianProduct{\alpha}{A}{X} \to (X_\alpha, \T_\alpha)$
	denote the $\alpha-$\ProjectionMap.
    We call the \WeakTopology
    on 
    \scCartesianProduct{\alpha}{A}{X}
    induced by 
    $\left\{\pi_\alpha\right\}_{\alpha \in A}$
    the \ProductTopology. 
\end{df}
