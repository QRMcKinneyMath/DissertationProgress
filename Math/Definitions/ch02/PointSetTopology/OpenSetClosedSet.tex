}
\begin{df}[\SetOpen, \SetClosed]
    Let $(X, \T)$ 
    be a 
    \TopologicalSpaceRef, 
    and let $A \in \T$. 
    We say that 
    $A$ is 
    \SetOpen
    in $(X,\T)$
    ( or \SetOpen in $X$ or \SetOpen in $\T$ or simply \SetOpen in cases where confusion
    won't result)
    and that $A$ posesses 
    \SetOpenness. 
    We say that 
    $X \setminus A$ 
    is 
    \SetClosed
    and that $X \setminus A$ 
    posesses 
    \SetClosedness
\end{df}
