\newcommand{\SetClosedness}[0]{\textbf{\hyperref[def:OpenSetClosedSet]{Closedness}}\xspace}
\begin{df}[\SetOpen, \SetClosed]
\label{def:OpenSetClosedSet}

\rm
    Let $(X, \T)$ 
    be a 
    \TopologicalSpace, 
    and let $A \in \T$. 
    Then we say that 
    $A$ is 
    \SetOpen
    in $(X,\T)$
    and we say that 
    $X \setminus A$ 
    is 
    \SetClosed
    in $(X,\T)$.
    When confusion won't result we may simply say 
    that A is \SetOpen in $X$ or in $\T$, or simply that $A$ is \SetOpen.
\end{df}
