\begin{df}[\NeighborhoodBasis]
\label{def:NeighborhoodBasis}

\rm
    Let $(X,\T)$ be a
    \TopologicalSpace
    and let $x \in X$. 
    Let 
    $F \subset \T$ 
    such that
    $\scNested{U}{\scU_{\scT}(x)}$.
    Further, let $x \in G$ for each $G \in F$. 
    Then we call $F$ a 
    \NeighborhoodBasis
    for $\T$ at $x$. 
\end{df}
