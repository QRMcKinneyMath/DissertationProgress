\begin{df}[\AtomicFormulae]
\label{def:AtomicFormula}

\rm
    Let $b$ be a \PredicateLetter{n} in \scA. 
    For $1 \leq i \leq n$, let $x_i$ be a \Term in \scG. 
    Then we call 
    \begin{equation*}
        b(x_1,x_2,\cdots,x_n)
    \end{equation*}
    an \AtomicFormula in \scG. 
\end{df}
