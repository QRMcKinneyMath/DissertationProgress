\begin{df}[\Term \TermAppearsInAtomicFormula \AtomicFormula]
\label{def:TermAppearsInAtomicFormula}

\rm
    Let $x$ be a \Term in $\scG$ 
    Let $y \StringEqual b(x_1,x_2,\cdots,x_n)$ be an \AtomicFormula in $\scG$. 
    We say that 
    $x$
    \TermAppearsInAtomicFormula
    $y$ 
    if for at least 1 i, $1 \leq i \leq n$, 
    we have $x$ \TermAppearsInTerm $x_i$. 
\end{df}
