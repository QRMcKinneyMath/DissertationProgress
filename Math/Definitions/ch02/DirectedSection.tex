\begin{df}[\DirectedSection of a \DirectedSet]
\label{def:DirectedSection}
\rm
    Let $(X, \leq)$ be a 
    \DirectedSet. 
    Let $x \in X$. 
    We define 
    \begin{equation*}
    S(x,\leq) = \{y \in X : x \leq y\}
    \end{equation*}
    We call $S(x,\leq)$ the 
    \DirectedSection of 
    $\leq$ corresponding to $x \in X$. 
\end{df}
