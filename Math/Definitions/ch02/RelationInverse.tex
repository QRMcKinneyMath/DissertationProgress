\newcommand{\RelationInverse}[0]{
    \textbf{\hyperref[def:RelationInverse]{Inverse}}
}
\begin{df}[\RelationInverse]
    Let $X\neq \emptyset$ and
    $Y \neq \emptyset$. 
    Let $R$ be a 
    \Relation
    from $X$ to $Y$. 
    We define 
    \begin{equation*}
    R^{-1} = \{(y,x) \in Y\times X | (x,y) \in R\}
    \end{equation*}
    We call $R^{-1}$ the 
    \RelationInverse
    of $R$. 
\end{df}
