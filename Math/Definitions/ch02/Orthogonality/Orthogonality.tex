\begin{df}[\Orthogonal]
\label{def:Orthogonal}
\rm
Let $X$ be a \SeminormedSpace over $\F$.
Let $x_0, y_0 \in X$. 
We say that $x_0$ is \Orthogonal to $y_0$ and we write $x_0 \perp y_0$. 
if, for each $\lambda \in \F$, we have
\begin{equation*}
\norm{x_0} \leq \norm{x_0+\lambda y_0} 
\end{equation*} 
\end{df}
