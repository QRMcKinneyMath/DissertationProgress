    \bf \hyperref[def:TopologicalGroup]{Topological Group}  \rm`
}
\begin{df}[\TopologicalGroup]
    Let $(G,+,e)$ be a \Group.
    \TopologyRef on
    $G$ such that 
    $+:G \times G \to G$ is \Continuous
    with respect to the 
    %\ProductTopology 


    and $g_{-1}$ is \Continuous.

    In this scenario, we call $(G, \T)$ a \TopologicalGroup.
\end{df}
