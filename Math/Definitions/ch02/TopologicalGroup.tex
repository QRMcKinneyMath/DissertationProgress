\begin{df}[\TopologicalGroup]
    Let $(G,+,e)$ be a \Group.
    \Topology on
    $G$ such that 
    $+:G \times G \to G$ is 
    \ContinuousFunction
    with respect to the 
    %\ProductTopology 


    and $g_{-1}$ is 
    \ContinuousFunction.

    In this scenario, we call $(G, \T)$ a \TopologicalGroup.
\end{df}
