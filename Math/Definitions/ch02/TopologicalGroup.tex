\newcommand{\TopologicalGroup}[0]{ \textbf{\hyperref[def:TopologicalGroup]{Topological Group}}\xspace}
\begin{df}[\TopologicalGroup]

\rm
\label{def:TopologicalGroup}
    Let $(G,+,e)$ be a \Group.
    Let $\T$ be a 
    \Topology on
    $G$ such that 
    $+:G \times G \to G$
	and 
    $\scGroupInverseOperator_G$
    are both
    \ContinuousFunction
    with respect to the 
	\ProductTopology.
    Then we call $(G, \T)$ a \TopologicalGroup.
\end{df}
