\newcommand{\ConvexSet}[0]{
    \bf \hyperref[def:ConvexSet]{Convex} \rm 
}

\begin{df}[\ConvexSet]
    Let $V$ 
    be a 
    \VectorSpace
    over a $\Field$
    $\F \in \{\R, \C\}$. 
    Let 
    $K \subset \F$. 
    We say that 
    $K$ is
    \ConvexSet
    if 
    for every pair $x,y \in K$, 
    we have $[x,y] \subset K$.
\end{df}
