\newcommand{\ScalarHomogeneity}[0]{\textbf{\hyperref[def:scalarhomogeneous]{Scalar Homogeneity}}\xspace}
\begin{df}[\ScalarHomogeneous]
\label{def:scalarhomogeneous}
\rm
    Let $V$ and $U$ be 
	\VectorSpaces over a 
	\Field	$\F \in \{\R, \C\}$. 
    Let $p:V \to U$  such that 
    for each $\alpha \in \F$ and each $x \in V$,
    \begin{equation*}
        p(\alpha x) = \alpha p(x)
    \end{equation*}
    Then we say $p$
    is \ScalarHomogeneous and 
    $p$  posesses \ScalarHomogeneity.
\end{df}



\newcommand{\AbsScalarHomogeneous}[0]{\textbf{\hyperref[def:absolutevaluescalarhomogeneous]{Absolutely Scalar Homogeneous}}\xspace}
\newcommand{\AbsScalarHomogeneity}[0]{\textbf{\hyperref[def:absolutevaluescalarhomogeneous]{Absolute Scalar Homogeneity}}\xspace}

\begin{df}[\AbsScalarHomogeneous]
\label{def:absolutevaluescalarhomogeneous}
\rm
    Let $V$ and $U$  be a \VectorSpaces over a \Field $\F \in \{\R, \C\}$. 
    Let $p:V \to U$ such that 
    for each $\alpha \in \F$ and each $x \in V$, 
    \begin{equation*}
        p(\alpha x) = \abs{\alpha} p(x)
    \end{equation*}
    Then we say $p$ is 
    \AbsScalarHomogeneous and 
    we say
    p posesses \AbsScalarHomogeneity.
\end{df}


\begin{rmk}
\label{rmk:seminorm}
\rm
If V is a 
\VectorSpace over a 
\Field 
$\mathbb{F} \in \{\R, \C\}$, then for each 
$x \in V$,
 $0x=0$.
Hence, if $p$ is an \AbsScalarHomogeneous operator on $V$, then for any $x \in V$
\begin{equation*}
p(0)=p(0x)=|0|p(x)=0p(x)=0
\end{equation*}
If instead $p$ is \ScalarHomogeneous operator on $V$, then we have
\begin{equation*}
p(0)=p(0x)=0p(x)=0
\end{equation*}
that is, in either case,  $p(0)=0$. 
\end{rmk}
