\begin{df}[\VectorSubspace]
\label{def:VectorSubspace}
\rm
Let $V$ be a \VectorSpace over a \Field $\scF$.
Let $U \subset V$ such that $span(U) \subset U$. 
Then we call $U$ a \VectorSubspace of $V$. 
\end{df}
