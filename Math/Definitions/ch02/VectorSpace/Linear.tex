\newcommand{\Linearity}[0]{\textbf{\hyperref[def:linear]{Linearity}}\xspace}

\begin{df}[\Linear]
\label{def:linear}
\rm
    Let V, U be  
    \VectorSpaces
    over a 
    \Field
    $\F$. 
    Let $T:V \to U$ 
    a \ScalarHomogeneous
    \MagmaHomomorphism.
    Then
    we say $T$
    is \Linear.
\end{df}
