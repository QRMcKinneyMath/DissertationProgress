\newcommand{\KronekerDelta}[0]{\textbf{\hyperref[def:KronekerDelta]{Kroneker-\ensuremath{\delta}}}\xspace}
\begin{df}[\KronekerDelta]
\label{def:KronekerDelta}
\rm
Let $(\scR,+, \cdot, 0_{\scR}, 1_{\scR})$. be a \Ring. 
Define $\delta:\bbN \times \bbN \to \scR$ by 
\begin{equation*}
\delta(i,j) = \begin{dcases}
1_{\scR} & i = j\\
0_{\scR} & i \neq j
\end{dcases}
\end{equation*}
We call $\delta$ the \KronekerDelta function over $\scR$. 
When $\scR$ is clear, we call $\delta$ the \KronekerDelta function. 
We denote $\delta(i,j) = \delta_{i,j}$. 
\end{df}
