\begin{df}[\Span]
\label{def:Span}
\rm
Let $V$ be a \VectorSpace.
Let $K \subset V$. 
We define $span(K)$ to be the 
collection of all \LinearCombinations of elements of $K$. 
We call this set the \Span of $K$.
\end{df}
