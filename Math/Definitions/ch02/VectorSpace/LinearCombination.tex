\begin{df}[\LinearCombination]
\label{def:LinearCombaination}
\rm
Let $\scF$ be a \Field. 
Let $V$ be a \VectorSpace over $\scF$. 
Let $K \subset V$. 
Let $\{x_i\}_{i=1}^n \subset K$.
Let $\{\alpha_i\}_{i=1}^n \subset \scF$. 
Define 
\begin{equation*}
z= \sum\limits_{i=1}^n \alpha_i x_i
\end{equation*}
We call $z$ a 
\LinearCombination of elements of $K$ 
with coefficients $\{\alpha_i\}_{i=1}^n$. 
\end{df}
