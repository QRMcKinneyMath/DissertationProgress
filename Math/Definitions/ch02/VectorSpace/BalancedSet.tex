\begin{df}[\Balanced]
\label{def:BalancedSet}
\rm
    Let $V$ be a \VectorSpace 
    over a \Field 
    $\F \in \{\R, \C\}$.
    Let $S \subset V$. 
    We call $S$ a \BalancedSet and
    we say that $S$ is 
    \Balanced if 
    for each
    $\alpha \in \F$ 
    with 
    $\abs{\alpha} \leq 1$
    we have 
    $\alpha S \subset S$. 
\end{df}
