\label{def:LocalBasis}
\newcommand{\LocalBasis}[0]{
    \bf \hyperref[def:LocalBasis]{Local Basis} \rm
}
\begin{df}[\LocalBasis]
    Let $(G,\T)$ be a 
    \TopologicalGroup
    with \IdentityElement $e$.
    We call a 
    \NeighborhoodBasis of $\T$ about $e$
    a $\LocalBasis$ for $(G,\T)$. 
\end{df}
