\begin{df}[\SchauderBasis]
\label{def:SchauderBasis}
\rm
Let $X$ be a \TVS over
a \Field $\F$. 
Let $\{x_i\}_{i \in \N} \subset X$ such that
for each $x \in X$, there exists a unique 
$\{\alpha_i\}_{i \in \N} \subset \F$ such that 
$x = \sum\limits_{i \in \N} \alpha_i x_i$.
Then we call $\{x_i\}_{i \in \N}$ a \SchauderBasis
for $X$.
\end{df}
