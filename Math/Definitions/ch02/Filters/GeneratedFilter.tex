\begin{df}[\FilterSubbasis]
\label{def:GeneratedFilter}\rm
    Let $X$ be a nonempty set.
    Let $A \subset X$ such that 
    \begin{enumerate}[label=(\roman*), ref={\ref{def:GeneratedFilter}.~\roman*}]
       \item \label{def:FilterSubbase:IsNotEmpty} $\emptyset \neq A$. 
       \item \label{def:FilterSubbase:FiniteIntersectionsNonempty} 
       Finite intersections of elements of $A$ are nonempty. 
    \end{enumerate}
    Let $K$ denote the collection of finite intersections of elements of $A$. 
    Define 
    \begin{equation*}
        \scK=\left\{ U \cup P : U \in K \tab[0.25cm] and \tab[0.25cm] P \subset X \right\}
    \end{equation*}
    We call $\scK$ the \Filter
    on $X$
    \FilterGeneratedBy
    $A$. 
    and we call $A$ a 
    \FilterSubbasis
    for $G$. 
    By \ref{prop:FilterExistence}, 
    $\scK$ is in fact a \Filter, and is the 
    \CoarsestFilter \Filter on $X$ containing $A$. 
\end{df}
