\begin{df}[\FilterBase]
    Let $X \neq \emptyset$. 
    Let $\emptyset \neq \scB \subset \scPowerSet{X}$
    such that 
    \begin{enumerate}
        \item $\emptyset \not \in \scB$. 
        \item If $\{U,V\} \subset \scB$ with $U \cap V \neq \emptyset$, then there exists
        a $W \in \scB$ with $W \subset U \cap V$. 
    \end{enumerate}
    Then we call 
    $\scB$ 
    a
    \FilterBase
    on $X$. 
\end{df}
