\newcommand{\TopologyOfUniformConvergence}[0]{\textbf{\hyperref[def:TopologyOfUniformConvergence]{Topology of Uniform Convergence}}\xspace}
\begin{df}[\TopologyOfUniformConvergence]
\label{def:TopologyOfUniformConvergence}
\rm
    Let $X$ be a nonempty set. 
    Let $(Y,\scT_Y)$ be a \TopologicalGroup.
    Let $\scB$ be a \LocalBasis for $Y$. 
    Let $\scF$ be a 
    \Subgroup of the set of 
    functions  $T:X \to Y$. 
    Suppose
    $\emptyset \neq \scG \subset 2^X$  such that 
	$(\scG, \subset)$ is a \DirectedSet.
    For each $x \subset X$ and $y \subset Y$, and  define 
    $M(x, y) = \{f \in \scF : f(x) \subset y\}$.
    \begin{equation*}
    \scB_{\scF} = \{M(x,y) : x \in \scB \tab[0.25cm] and \tab[0.25cm] y \in \scB\}
    \end{equation*}
    For each $f \in \scF$, define 
    \begin{equation*}
    \scB(f) = f+\scB_{\scF}
    \end{equation*}
By \ref{prop:TopologyOfUniformConvergence}, 
there is a unique left-invariant \Topology $\scT$ on 
$\scF$ such that for each $f \in \scF$, 
$\scB(f)$ is a 
\FilterBase for the \NeighborhoodFilter of $\scT$ 
at $f$. 
We call $\scT$ the 
\TopologyOfUniformConvergence induced by $\scG$ on $\scF$. 
\end{df}
