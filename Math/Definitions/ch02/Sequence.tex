\newcommand{\Sequences}[0]{\textbf{\hyperref[def:Sequence]{Sequences}}\xspace}
\begin{df}[\Sequence]
\label{def:Sequence}
\rm
    Let $X$ be a set. 
    A \Sequence in $X$ is a 
    \Function $f:\N \to X$. 
    If $f$ is a \Sequence 
    in $X$ and 
    $f(n) = x_n$ for $n \in \N$, then 
    we may refer to $\{x_n\}_{n \in \N}$ as the \Sequence itself. 
\end{df}
