 \newcommand{\QuotientMapInstance}[3]{ #1 : #2\to #2/#3 }
\begin{df}[\QuotientMap]
\label{df:quotient_map}
\rm
    Let $X$ be a nonempty set.
    Let $\cong$ be an 
	\EquivalenceRelation 
	on X.
    \LetBeQuotientSet{X}{\cong}
    Define $T:X \to X/\cong$ by setting, for each $x \in X$, 
    \begin{equation*}
        T(x)=[x]
    \end{equation*}    
    We call T the \QuotientMap of X under $\cong$. 
	If $A \subset X$ 
	and $T^{-1}(T(A)) = A$, then we call $A$ 
	a \Fiber under $\cong$.
    If $\cong$ is understood, we call 
	$A$ a \Fiber.
\end{df}
