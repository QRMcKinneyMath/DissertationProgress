\begin{df}[Quotient Map]

    Let $X \neq \emptyset$.
    Let $\cong$ be an 
	\EquivalenceRelation 
	on X.
    \LetBeQuotientSet{X}{$\cong$}
    Define $T:X \to X/\cong$ by setting, for each $x \in X$, 
    \begin{equation}
        T(x)=[x]
    \end{equation}    
    We call T the \QuotientMap of X under $\cong$. 
\end{df}
