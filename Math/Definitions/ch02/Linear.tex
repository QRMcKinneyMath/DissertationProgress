}
\newcommand{\Linearity}[0]{
    \bf \hyperref[def:linearity]{Linearity} \rm
}

\begin{df}[\Linear]
    Let V, U be  
    \VectorSpaces
    over a 
    \Field
    $\F$. 
    We say that 
    $T:V \to U$ 
    is \Linear
    or that T possesses 
    \Linearity
    if T is both 
    \Additive
    and \ScalarHomogeneous
\end{df}
