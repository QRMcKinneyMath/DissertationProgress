\newcommand{\Toset}[0]{
    \bf \hyperref[def:TotalOrder]{Totally Ordered Set} \rm
}
\begin{df}[\TotalOrder]
    Let $(X,R)$ be a 
    \Poset in which
    every pair of elements is 
    \Comparable. 
    Then we call 
    $R$ a 
    \TotalOrder
    on $X$
    and we call 
    $(X,R)$ a 
    \Toset.
\end{df}
