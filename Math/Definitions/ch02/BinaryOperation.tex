\begin{df}[\BinaryOperation]
    Let $X \neq \emptyset$
    be a set. 
    We call 
    $T:X \times X \to X$ a 
    \BinaryOperation
    on $X$. 
    In this context, 
    for $x,y \in X$, 
    we sometimes use the notation
    \begin{equation*}
        xTy=T(x,y)
    \end{equation*}
\end{df}
