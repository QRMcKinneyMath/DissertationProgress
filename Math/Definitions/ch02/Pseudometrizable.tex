\newcommand{\PseudometricCompatible}[0]{\textbf{\hyperref[def:Pseudometrizable]{Compatible}}\xspace}

\begin{df}[\Pseudometrizable]
    \label{def:Pseudometrizable}
    \rm
    Let $(X,\T)$ be a \TopologicalSpace.
	Let $\tilde{d}$ be a \Pseudometric on $X$. 
    \begin{enumerate}
		\item We say that $\T$ and $\tilde{d}$ are \PseudometricCompatible if 
		$\scT$ is the \PseudometricInducedTopology on $(X,\tilde{d})$. 
        \item We say that $(X,\T)$ is \Pseudometrizable if there exists a \Pseudometric  $d$
		on $X$ which is \PseudometricCompatible with $d$.
        \item We say that $(X,\T)$ is \Metrizable if there exists a \Metric $d$
		on $X$ which is \PseudometricCompatible with $d$.
    \end{enumerate}
\end{df}
