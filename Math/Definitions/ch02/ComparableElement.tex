\newcommand{\Comparability}[0]{\textbf{\hyperref[def:Comparable]{Comparability}}\xspace}

\begin{df}[\Comparable]
   Let $(X,R)$ be a 
   \PreorderedSet. 
   We say that $x,y \in X$ are 
   \Comparable and that
   they possess 
   \Comparability
   if 
   $xRy$ or $yRx$. 
\end{df}
