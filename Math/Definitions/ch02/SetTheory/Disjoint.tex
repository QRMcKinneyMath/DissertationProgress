\begin{df}[Disjoint]
\label{def:Disjoint}

\rm
    Let $X$ and $Y$ be sets such that 
    $X \cap Y = \emptyset$. 
    Then we say that $X$ and $Y$ are 
    \Disjoint. 
    Let $F=\{X_{\alpha}\}_{\alpha \in A}$ 
	be a collection of sets
    such that for each $\alpha, \beta \in A$ 
    with $\alpha \neq \beta$, we have 
    $X_\alpha$ 
	and 
	$X_\beta$
    are \Disjoint.
    Then we say that $F$ is \Disjoint. 
\end{df}
