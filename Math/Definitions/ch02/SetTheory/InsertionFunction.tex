\newcommand{\InsertionFunctions}[0]{
    \textbf{\hyperref[def:InsertionFunction]{Insertion Functions}}
}\begin{df}[Insertion Function]
\label{def:InsertionFunction}

\rm
    Let $A \subset B$ and define 
    $f:A \to B$ by $f(x)=x$. 
    The we call $f$ the 
    \InsertionFunction of $A$ into $B$. 
\end{df}
