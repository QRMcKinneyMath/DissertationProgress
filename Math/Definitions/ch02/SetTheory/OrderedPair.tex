\begin{df}[\OrderedPair]
    Let $X$ and $Y$ be sets. 
    Then we define 
    $(X,Y)= \{\{X\}, \{X, Y\}\} \in \scPowerSet{\{X, Y\}}$. 
    We call $(X,Y)$ the \OrderedPair of $X$ with $Y$. 
\end{df}
