\begin{df}[Ordered Pair]
\label{def:OrderedPair}

\rm
    Let $X$ and $Y$ be sets. 
    Then we define 
    $(X,Y)= \{\{X\}, \{X, Y\}\} \in 2^{\{X, Y\}}$. 
    We call $(X,Y)$ the \OrderedPair of $X$ with $Y$. 
\end{df}
