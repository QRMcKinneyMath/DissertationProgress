\begin{df}[\Equal]
\label{def:SetEquality}

\rm
    Let $X$ and $Y$ be \Sets. 
    We say that $X$ and $Y$ are 
    \Equal, and we write
    $X = Y$ if
    \begin{equation*}
        \pa{\forall \phi}\pa{\phi \in X \iff \phi \in Y}
    \end{equation*}



\end{df}
