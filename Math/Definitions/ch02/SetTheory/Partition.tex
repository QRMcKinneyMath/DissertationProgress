\begin{df}[Partition]
\label{def:Partition}
\rm
    Let $X$ be a set and 
    $Y \subset \scPowerSet{X}$ 
    be a \Disjoint \Cover of $X$. 
    Then we call $Y$ a \Partition
    of $X$. 
\end{df}
