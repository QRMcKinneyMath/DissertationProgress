\begin{df}[Binary Cartesian Product]
\label{def:BinaryCartesianProduct}

\rm
    Let $X \neq \emptyset$ and let 
    $Y \neq \emptyset$. 
    We define 
    $X \times Y =\{(x,y) \in \scPowerSet{\scPowerSet{X \cup Y}} | x \in X \wedge y \in Y\}$
    We call 
    $X \times Y$ the 
    \BinaryCartesianProduct
    of $X$ with $Y$. 
\end{df}
