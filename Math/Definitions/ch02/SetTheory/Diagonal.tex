\begin{df}[Set Diagonal]
\label{def:SetDiagonal}

\rm
    Let $X$ be a set. 
    We define 
    $\scSetDiagonal{X}=\{(x,x)\in X \times X | x \in X\}$
    and we call 
    \scSetDiagonal{X} 
    the 
    \SetDiagonal
    of $X$.
\end{df}
