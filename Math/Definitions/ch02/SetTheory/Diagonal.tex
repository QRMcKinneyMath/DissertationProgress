\newcommand{\scSetDiagonal}[1]{
    \ensuremath{\hyperref[def:SetDiagonal]{\Delta\pa{#1}}}
}
\begin{df}[\SetDiagonal]
    Let $X$ be a set. 
    We define 
    $\scSetDiagonal{X}=\{(x,x)\in X \times X | x \in X\}$
    and we call 
    \scSetDiagonal{X} 
    the 
    \SetDiagonal
    of $X$.
\end{df}
