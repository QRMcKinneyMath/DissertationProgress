\begin{df}[\Surjection]
\label{def:Surjective}

\rm
   Let $X$ and $Y$ be nonempty sets.
   Let $f:X \to Y$. 
   Suppose that 
   for each $y \in Y$, 
   there exists an 
   $x \in X$ such that 
   $f(x) = y$. 
   Then we say that $f$ 
   is a
   \Surjection onto $Y$, 
   and we call $f$ 
   \Surjective
   onto $Y$. 
\end{df}
