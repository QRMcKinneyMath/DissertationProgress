}
\newcommand{\Surjection}[0]{
    \bf \hyperref[def:Surjective]{Surjection} \rm
}
\begin{df}[\Surjective]
   Let $X,Y$ be sets and let 
   $f:X \to Y$. 
   Suppose that 
   for each $y \in Y$, 
   there exists an 
   $x \in X$ such that 
   $f(x) = y$. 
   Then we say that $f$ 
   is a
   \Surjection, 
   and we call $f$ 
   \Surjective. 
\end{df}
