\begin{df}[\Nested]
\label{def:NestingCondition}
    Let $F,G \neq \emptyset$. 
    We say that \scNested{F}{G} holds if, 
    for each $g \in G$, there exists $f \in F$ 
    such that $f \subset g$. 
\end{df}
