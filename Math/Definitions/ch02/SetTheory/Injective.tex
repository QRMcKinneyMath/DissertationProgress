}\begin{df}[Injective]
\label{def:Injective}

\rm
    Let $X,Y$ be sets and let 
    $f:X \to Y$. 
    We say that $f$ is 
    an 
    \Injection, 
    or that $f$ is 
    \Injective if 
    for all $x,y \in X$, 
    if $x \neq y$, then 
    $f(x) \neq f(y)$. 
\end{df}
