\begin{df}[Bijective]
\label{def:Bijective}

\rm
    Let $X$ and $Y$ be sets and let 
    $f:X \to Y$ be \Surjective and \Injective. 
    Then we say that $f$ is 
    \Bijective, or we say that f is a 
    \Bijection. 
\end{df}
