\begin{df}[Subset]
\label{def:Subset}

\rm
    Let $X$ and 
    $Y$ be sets
    such that $x \in Y\implies x \in X$. 
    Then we write $Y \subset X$ 
    and we say that $Y$ is a 
    \QSubset of $X$
    and we write $X \supset Y$
    and we say that $X$ is a 
    \Superset of $Y$. 
    
\end{df}
