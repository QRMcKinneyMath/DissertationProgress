\newcommand{\TvsDuals}[0]{\textbf{\hyperref[def:TvsDualSpace]{Topological Duals}}\xspace}

\begin{df}[\TvsDual]
\label{def:TvsDualSpace}
\rm
Let $X$ be a \TVS. 
Let $X'$ denote the \AlgebraicDual of $X$. 
We denote with $X^*$ the subspace of $X'$ consisting of 
\ContinuousFunction functionals.
We call 
$X^*$ the \TvsDual of $X$. 
\end{df}
