\begin{rmk}[A-\UniformNeighborhoods]
\label{rmk:UniformNeighborhood}
Let $(X,\scW, \scT_{\scW})$ be a \UniformSpace. 
Let $A \subset X$ and 
let $U$ be a \SymmetricRelation \Entourage.
By \ref{prop:UniformTopology:Neighborhood}, 
$V(A)$ is a \Neighborhood of A. 
We shall now refer to this set as a $V-\UniformNeighborhood$ of A.
\end{rmk}
