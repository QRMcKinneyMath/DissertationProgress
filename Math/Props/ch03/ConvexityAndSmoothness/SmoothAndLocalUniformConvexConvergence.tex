\begin{prop}[Normalized Convergence]
\label{prop:NormalizedConvergence}
\rm
Let $X$ be a \SmoothSpace, 
\LocallyUniformlyConvex \SeminormedSpace.
Let $J$ denote the \NormalizedDualityMap of $X$. 
Let $\{x_n\}_{n \in \N} \subset X$.
Let $x_0 \in X$.
suppose $\ip{x_n-x_0, Jx_n-Jx_0} \to 0$. 
Then $x_n \to x_0$. 
\begin{equation} 
\ip{x_n-x_0, Jx_n-Jx_0} \to 0 \implies x_n \to x_0
\end{equation}

\end{proof}
\end{prop}
