\begin{prop}[\QuotientMap is  \Surjective]
    Let $X \neq \emptyset$. 
    Let $\cong$ be an 
	\EquivalenceRelation on 
	$X$.
    Let $\QuotientMapInstance{T}{X}{\cong}$  be the 
	\QuotientMap of $X$ under the 
	\Relation
	$\cong$. 
    Then T is a 
	\Surjection. 
    \begin{proof}
       Let $K \in Z/\cong$. 
       Then for some $x \in Z$, $K=[x]$. 
       Then $T(x) = K$. 
       Since K was arbitrary, we are done. 
    \end{proof}
\end{prop}
