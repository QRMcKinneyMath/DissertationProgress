\begin{prop}[\CommutativeFunction \Subgroups are \NormalSubgroups]
\label{prop:CommutativeSubgroupNormal}
\rm
Any \Subgroup of a \CommutativeGroup is a \NormalSubgroup.
\begin{proof}
Let $K$ be a \Subgroup of a \CommutativeFunction \Group $G$. 
Let $x \in K$ and $y \in G$. 
Then $yxy^{-1}=yy^{-1}x=x \in K$. 
\end{proof}
\end{prop}
