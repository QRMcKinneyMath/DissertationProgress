\begin{prop}[\ScalingOperator]
    \label{prop:ScalingOperatorAlgebraicProperties}
    \rm
    Let $V$
    be a 
    \VectorSpace
    over a 
    \Field
    $\F$. 
    Let $\alpha,\beta \in \F$. 
    Then $M_\alpha \circ M_\beta = M_{\alpha * \beta}$. 
    \begin{proof}
        Let $v \in V$. Then 
        \begin{align*}
            M_{\alpha} \circ M_{\beta} v & = M_\alpha \pa{\beta * v } \\
            & = \alpha * (\beta * v) \\
            & = (\alpha * \beta) * v \\
            & = M_{\alpha*\beta}v
        \end{align*}
    \end{proof} 
\end{prop}
