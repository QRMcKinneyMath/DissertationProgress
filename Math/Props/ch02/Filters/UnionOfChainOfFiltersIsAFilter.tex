\begin{prop}[Union/Intersection of chain of filters is a filter ]
    \label{prop:UnionIntersectionChainOfFiltersIsAFilter}
    Let \(X \neq \emptyset\) and 
    \(\{ \scF_\alpha \}_{\alpha \in A}\) be a 
    \Chain of \Filters on $X$. 
    Then 
    \begin{enumerate}[label=(\roman*), ref={\ref{prop:UnionIntersectionChainOfFiltersIsAFilter}~\roman*}]
        \item \label{prop:UnionOfChainOfFiltersIsFinerFilter}$\bigcup\limits_{\alpha \in A} \scF_\alpha$ is a \Filter on $X$ which is \FinerFilter than each $X_\alpha$. 
        \item \label{prop:IntersectionOfChainOfFiltersIsFinerFilter} $\bigcap\limits_{\alpha \in A}\scF_\alpha$ is a \Filter on $X$ which is \CoarserFilter than each $X_\alpha$. 
    \end{equation}
