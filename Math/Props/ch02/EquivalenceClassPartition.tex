\begin{prop}[Equivalence Classes Partition]
    \label{prop:EquivalenceClassesPartition}
\rm
    Let $X$ be a nonempty set.
    Let $\cong$ be an 
	\EquivalenceRelation
	defined on X. 
    Let $x,y \in X$. 
    The following statements are equivalent. 
    \begin{enumerate}[label=(\roman*), ref={\ref{prop:EquivalenceClassesPartition}.~\roman*}]
        \item 
		\label{prop:EquivalenceClass:NonDisjoint}
		$[x]_{\cong}  \cap [y]_{\cong} \neq \emptyset$
        \item 
		\label{prop:EquivalenceClass:Related}
		$x \cong y$
        \item 
		\label{prop:EquivalenceClass:EqualClass}
		$[x]_{\cong} = [y]_{\cong}$
        \item 
		\label{prop:EquivalenceClass:Subset}
		$[x]_{\cong} \subset [y]_{\cong}$.
    \end{enumerate}
    
\begin{proof}[\ref{prop:EquivalenceClass:NonDisjoint} $\implies$ \ref{prop:EquivalenceClass:Related}]
Suppose $[x]_{\cong} \cap [y]_{\cong} \neq \emptyset$. 
Then there exists $z \in [x]_{\cong} \cap [y]_{\cong}$.
Hence $z \cong x$ and $z \cong y$. By \RelationSymmetry, $x \cong z$. 
By \RelationTransitivity, $x \cong y$.
\end{proof}
\begin{proof}[\ref{prop:EquivalenceClass:Related} $\implies$ \ref{prop:EquivalenceClass:Subset}]
    Let $x \cong y$ and let $z \in [x]_{\cong}$. 
    Then $z \cong x \cong y$, so $z \cong y$ and $z \in [y]_{\cong}$.
    Since z was arbitrary, we're done. 
\end{proof}
\begin{proof}[\ref{prop:EquivalenceClass:Related} $\implies$ \ref{prop:EquivalenceClass:EqualClass}]
    Let $x \cong y$. 
    By 
    $\ref{prop:EquivalenceClass:Related} \implies \ref{prop:EquivalenceClass:Subset}$, 
    $[x]_{\cong} \subset [y]_{\cong}$. 
    Also, by \RelationSymmetry, $y \cong x$, 
    so $[y]_{\cong} \subset [y]_{\cong}$.
    Hence, equality holds.
\end{proof}
\begin{proof}[\ref{prop:EquivalenceClass:Subset} $\implies$ \ref{prop:EquivalenceClass:NonDisjoint}]
    Let $[y]_{\cong} \subset [x]_{\cong}$. 
    Then $y \in [y]_{\cong} = [y]_{\cong} \cap [x]_{\cong} $.
    Hence \ref{prop:EquivalenceClass:NonDisjoint} holds.
\end{proof}

\begin{proof}[\ref{prop:EquivalenceClass:EqualClass} $\implies$  \ref{prop:EquivalenceClass:Subset}]
Obvious.
\end{proof}
\end{prop}
