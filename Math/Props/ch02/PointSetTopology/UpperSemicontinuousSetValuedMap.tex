\begin{prop}[\UpperSemicontinuousSetValued]
\label{prop:UpperSemiSetValued}
Let $X$ be a \TopologicalSpace. 
Let $Y$ be a \PseudoHausdorff \TopologicalSpace.
Let $Q$ denote the \QuotientMap of $Y$.
Let $T:X \to 2^Y$.
Let $x_0 \in X$. 
Suppose $T(x_0)$ is \SetCompact. 
The following conditions are equivalent. 
\begin{enumerate}[label=(\roman*), ref={\ref{def:UpperSemiSetValued}~\roman*}]
\item
\label{prop:UpperSemi:Top}
$T$ is \UpperSemicontinuousSetValued at $x_0$.
\item
\label{prop:UpperSemi:Net}
For each \Net $\{x_\alpha\}_{\alpha \in A} \subset Y$ with $x_\alpha \to x_0$.
For each net $\{y_\alpha\}_{\alpha \in A} \subset Y$ with $y_\alpha \in T(x_\alpha)$, There exists a $y_0 \in T(x_0)$ such that $y_0$ is a cluster point of $\{y_\alpha\}_{\alpha \in A}$.
\end{enumerate}
