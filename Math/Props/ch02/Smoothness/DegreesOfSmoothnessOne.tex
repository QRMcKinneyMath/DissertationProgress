\begin{prop}[Degrees Of Smoothness]
\label{prop:DegreesOfSmoothnessOne}
\rm
Let $X$ be a \SeminormedSpace. The following are true.
\begin{enumerate}[label=(\roman*), ref={\ref{prop:DegreesOfSmoothnessOne}~\roman*}]
\item 
\label{prop:DegreesOfSmoothness:UniformImpliesLocal}
If $X$ is \UniformlySmoothSpace, then it is \LocallyUniformlySmoothSpace.
\item 
\label{prop:DegreesOfSmoothness:LocalImpliesSmooth}
If $X$ is \LocallyUniformlySmoothSpace, at $x_0$, then it is \SmoothSpace at $x_0$.
\end{enumerate}

\begin{proof}[Proof of \ref{prop:DegreesOfSmoothness:UniformImpliesLocal}]
This is clear directly from 
\ref{def:ModulusOfSmoothness}, 
\ref{def:LocallyUniformlySmoothSpace}, 
and 
\ref{def:UniformlySmoothSpace}.
\end{proof}
\begin{proof}[Proof of \ref{prop:DegreesOfSmoothness:LocalImpliesSmooth}]
I prove by converse. 
Let $X$ be not \SmoothSpace at $x_0 \in X$. 
Then there exists $j_1,j_2 \in J(x_0)$ with $j_1 \neq j_2$. 
Since $\ip{x_0, j_1} = \ip{x_0, j_2}$, there does not exists $\alpha \in \F$
such that $\alpha j_1= j_2$.
This implies $j_1 \not \in  span(j_2)$.
Hence, by 
\ref{prop:FunctionalKernelIntersection}, 
we conclude $Kernel(j_1) \neq Kernel(j_2)$. 
Hence we can construct $y \in X$ such that 
$\norm{y} = \norm{x_0}$, 
$\ip{y, j_1} = -\ip{y, j_2} > 0$. 
Then $y \in Kernel(j_1+j_2)$. 
Then, for each $t > 0$, we have 
\begin{align*}
0 & < t \ip{y, j_1} \\
& = \frac{t}{2} \ip{y, j_1} - \frac{t}{2} \ip{y, j_2}\\
& =\frac{1}{2} \pa{ \pa{\norm{x_0}^2 + t \ip{y, j_1}} + \pa{ \norm{x_0}^2 - t \ip{y, j_2}}-2 \norm{x_0}^2}\\
& =  \frac{1}{2} \pa{ \ip{x_0+ ty, j_1} + \ip{x_0 -ty, j_2} - 2 \norm{x_0}^2}\\
& \leq \frac{\norm{x_0}}{2} \pa{ \norm{x_0+ty} + \norm{x_0-ty} -2 \norm{x_0}}\\
& \leq \norm{x_0} \overline{\rho}(t, x_0)
\end{align*}
Hence, $0 < \frac{\ip{y, j_1}}{\norm{x_0}} \leq \frac{\overline{\rho(t,x_0)}}{t}$.
Since $t>0$ was arbitrary, $\lim\limits_{t \to 0} \frac{\overline{\rho(t,x_0)}}{t} > 0$, so 
that $X$ is not 
\LocallyUniformlySmoothSpace at $x_0$. 
\end{proof}
\end{prop}
