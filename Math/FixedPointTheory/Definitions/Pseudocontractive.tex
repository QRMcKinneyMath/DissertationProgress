\begin{df}[\Pseudocontractive]
\label{def:Pseudocontractive}
\rm
Let $X$ be a \SeminormedSpace. 
Let $K \subset X$. 
Let $T:K \to X$. 
Suppose that for each $x,y \in K$, and for each $\lambda > 0$, 
\begin{equation*}
\norm{x-y} \leq \norm{x-y + \lambda \pa{(I-T)x-(I-T)y}}
\end{equation*}
Then we say that $T$ is \Pseudocontractive.
\end{df}
