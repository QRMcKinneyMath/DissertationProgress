\begin{df}[\PseudoFixedPoint]
\label{def:PseudoFixedPoint}
\rm
Let $X$ be a \TVS.
Let $K \subset X$. 
Let $T:K \to K$. 
Let $x_0 \in K$
We say that $x_0+\overline{0}$ is a \PseudoFixedPoint of $T$ if 
$T\pa{\pa{x_0+\overline{0}} \cap K} \subset \pa{x_0+\overline{0}} \cap K$.
We denote the set of \PseudoFixedPoints of $T$ with $PF(T)$. 
\end{df}
