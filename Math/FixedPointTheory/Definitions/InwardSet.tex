\begin{df}[\InwardSet]
\label{def:InwardSet}
\rm
Let $X$ be a \VectorSpace over $\F \in \{\R, \C\}$. 
Let $K \subset X$. 
We define $I_k:K \to 2^X$ by 
$I_k(x) = \{x+ \lambda (v-x) : \pa{v \in K}\pa{\lambda \geq 1}\}$
We call $I_k(x)$ the \InwardSet of $K$ at $x$. 
\end{df}
