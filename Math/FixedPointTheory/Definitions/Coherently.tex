\newcommand{\Coherently}[0]{\textbf{\hyperref[def:Coherently]{Coherently}}\xspace}
\begin{df}[\Coherently]
\label{def:Coherently}
\rm
Let $\pa{X, \{\norm{\cdot}_i\}_{i \in \N}}$ be a \GSpace. 
Let $K \subset X$. 
Let $T:K \to X$. 
Let $A$ and $B$ be properties which may be possessed by 
a mapping from a subset of a \SeminormedSpace
into that same \SeminormedSpace. 
We say that $T$ \Coherently possesses properties A and B 
if there are infinitely many $n \in \N$
such that
$T:K \subset \pa{X, \norm{\cdot}_n} \to \pa{X, \norm{\cdot}_n}$ 
posses properties A and B. 
\end{df}
