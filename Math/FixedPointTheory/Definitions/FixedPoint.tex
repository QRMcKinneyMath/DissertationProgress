\newcommand{\FixedPoint}[0]{\textbf{\hyperref[def:FixedPoint]{Fixed Point}}\xspace}
\newcommand{\FixedPoints}[0]{\textbf{\hyperref[def:FixedPoint]{Fixed Points}}\xspace}
\begin{df}[\FixedPoint]
\label{def:FixedPoint}
\rm
Let $X$ be a set. 
Let $T:X \to X$. 
We say that $x$ is a \FixedPoint of $T$ if 
$Tx=x$. 
We denote the set of \FixedPoints of $T$ with $fix(T)$. 
\end{df}
